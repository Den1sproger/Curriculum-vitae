function showSumByRow(matrix) {
  console.log('show sum by row\n')

  for (let i = 0; i < matrix.length; i++) {
    let sum = 0
    for (let j = 0; j < matrix[i].length; j++) {
      sum += matrix[i][j]
    }
    console.log(`row ${i} => ${sum}`)
  }
}
